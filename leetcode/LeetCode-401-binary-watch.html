public List<String> readBinaryWatch(int num) {
        int[] bits = {1, 2, 4, 8, 16, 32};
        Set<String> res = new HashSet<>();
        for (int hourNum = 0; hourNum <= 4; hourNum++) {
            int minuteNum = num - hourNum;
            if (minuteNum < 0) {
                break;
            }
            List<Integer> hours = new ArrayList<>();
            dfs(hourNum, 0, 3, 0, hours, bits);

            List<Integer> minutes = new ArrayList<>();
            dfs(minuteNum, 0, 5, 0, minutes, bits);

            for (int hour : hours) {
                if (hour > 11) {
                    continue;
                }
                for (int minute : minutes) {
                    if (minute > 59) {
                        continue;
                    }
                    StringBuilder time = new StringBuilder();
                    time.append(hour);
                    time.append(":");
                    if (minute < 10) {
                        time.append("0");
                    }
                    time.append(minute);
                    res.add(time.toString());
                }
            }
        }
        return new ArrayList<>(res);
    }

    private void dfs(int num, int begin, int end, int sum, List<Integer> res, int[] bits) {
        if (num == 0) {
            res.add(sum);
            return;
        }
        for (int i = begin; i <= end; i++) {
            sum += bits[i];
            dfs(num - 1, i + 1, end, sum, res, bits);
            sum -= bits[i];
        }
    }
